// On-chain PDA derivation
pub fn create_agent_wallet(
  ctx: Context<CreateWallet>,
  agent_id: String,
  per_tx_limit: u64,
  daily_limit: u64,
) -> Result<()> {
  let wallet = &mut ctx.accounts.wallet;
  wallet.authority = ctx.accounts.authority.key();
  wallet.per_tx_limit = per_tx_limit;
  wallet.daily_limit = daily_limit;
  Ok(())
}
