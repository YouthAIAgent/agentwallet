use anchor_lang::prelude::*;
use crate::state::*;
use crate::error::*;

/// Emergency freeze: instantly halt ALL agent operations
/// Owner can freeze entire system in 1 tx if anomaly detected
#[derive(Accounts)]
pub struct EmergencyFreeze<'info> {
    #[account(mut, has_one = owner)]
    pub global_state: Account<'info, GlobalState>,
    pub owner: Signer<'info>,
}

pub fn emergency_freeze(
    ctx: Context<EmergencyFreeze>,
    reason: String,
) -> Result<()> {
    let global_state = &mut ctx.accounts.global_state;
    
    // INSTANT system-wide freeze
    global_state.is_frozen = true;
    global_state.freeze_timestamp = Clock::get()?.unix_timestamp;
    global_state.freeze_reason = reason;
    
    // Emit freeze event for monitoring
    emit!(FreezeEvent {
        timestamp: global_state.freeze_timestamp,
        reason: global_state.freeze_reason.clone(),
    });
    Ok(())
