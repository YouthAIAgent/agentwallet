// Agent Identity PDA derivation
pub fn derive_agent_identity(
    agent_pubkey: &Pubkey,
    program_id: &Pubkey,
) -> (Pubkey, u8) {
    Pubkey::find_program_address(
        &[
            b"agent_identity",
            agent_pubkey.as_ref(),
        ],
        program_id,
    )
}

// Identity account structure
pub struct AgentIdentity {
    pub agent: Pubkey,
    pub reputation: u64,
    pub tasks_completed: u32,
    pub created_at: i64,
}
